#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <signal.h>
#include <sys/wait.h>
#include <errno.h>

void restart_real_work(int sig);
void start_real_work();


int main()
{
    pid_t pid = fork();
    if (pid == -1)  {
        fprintf (stderr,"fork faild %s \n",strerror(errno));
        return 0;
    }
    if (pid == 0)   {
        start_real_work();
    } else {
        signal( SIGCHLD, restart_real_work);
        while (1) {
            sleep (6);
        }
    }
    return 0;
}

void start_real_work()
{
    /*屏蔽管道破裂信号*/
    signal(SIGPIPE, SIG_IGN);
    
    while(1) {
        printf("working");
        sleep(1);
    }
}

/*守护信号函数*/
void restart_real_work(int sig)
{
    fprintf (stderr,"守护进程\n");
    pid_t childPid = waitpid(-1, NULL, 0);
    if(childPid <= 0) {
        return;
    }
    pid_t pid = fork();
    if(pid == -1) {
        fprintf(stderr, "fork failed, errmsg %s\n", strerror(errno));
        return;
    }
    if(pid == 0) {
        start_real_work();
    }
    return ;
}
