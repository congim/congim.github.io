Docker启动mysql(5.7)并挂载本地配置文件、数据目录

December 02, 2021 a min to read Cong post

本地配置文件

vim /etc/mysql/conf/my.cnf


# For advice on how to change settings please see
# http://dev.mysql.com/doc/refman/5.7/en/server-configuration-defaults.html

[mysqld]
#
# Remove leading # and set to the amount of RAM for the most important data
# cache in MySQL. Start at 70% of total RAM for dedicated server, else 10%.
# innodb_buffer_pool_size = 128M
#
# Remove leading # to turn on a very important data integrity option: logging
# changes to the binary log between backups.
# log_bin
#
# Remove leading # to set options mainly useful for reporting servers.
# The server defaults are faster for transactions and fast SELECTs.
# Adjust sizes as needed, experiment to find the optimal values.
# join_buffer_size = 128M
# sort_buffer_size = 2M
# read_rnd_buffer_size = 2M
datadir=/var/lib/mysql
socket=/var/lib/mysql/mysql.sock

# Disabling symbolic-links is recommended to prevent assorted security risks
# sql_mode=ONLY_FULL_GROUP_BY,STRICT_TRANS_TABLES,ERROR_FOR_DIVISION_BY_ZERO,NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTION
sql_mode=STRICT_TRANS_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTION
symbolic-links=0
#default-character-set=utf8
lower_case_table_names=1
explicit_defaults_for_timestamp=1
max_connections=512
max_allowed_packet=10000M

log-error=/var/log/mysqld.log
pid-file=/var/run/mysqld/mysqld.pid

docker 启动

docker run -it -d --privileged=true -p 3306:3306  --restart=always  --name mysql -v /etc/mysql/conf:/etc/mysql -v /etc/mysql/mysqlData:/var/lib/mysql -e MYSQL_ROOT_PASSWORD="123456" mysql:5.7 --log-error=/var/log/mysql/error
